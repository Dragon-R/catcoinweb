//lista de usuarios
SELECT usuario.alias, usuario.nombre, usuario.genero, usuario.email FROM `usuario`;

//usuarios y metodos de pago
SELECT usuario.id_usuario, usuario.nombre, usuario.email, metodo_pago.id_pago, metodo_pago.nro_tarjeta, metodo_pago.tigo_money
FROM usuario
INNER JOIN metodo_pago
ON usuario.id_usuario = metodo_pago.id_usuario;

//usuarios con metodos  de pago y sin metodos de pago
SELECT usuario.id_usuario, metodo_pago.id_pago
FROM usuario
LEFT JOIN metodo_pago
ON usuario.id_usuario = metodo_pago.id_usuario;

//usuarios que jugaron algun juego
SELECT usuario.id_usuario, usuario.nombre, usuario.email, juego.id_juego, juego.descripcion, juego.fecha_inicio, juego.fecha_fin, juego.monto
FROM usuario
INNER JOIN juego
ON usuario.id_usuario = juego.id_usuario;

//usuarios que ganaron o perdieron
SELECT usuario.id_usuario, usuario.nombre, usuario.email, gana.vict_derr, gana.monto_gan
FROM usuario
INNER JOIN gana
ON usuario.id_usuario = gana.id_usuario;
